Precision-Recall Curve

# Assuming multinom_model is your multinomial logistic regression model

 

# Make predictions on test data to obtain predicted probabilities for each class

probabilities <- predict(multinom_model, newdata = test_data, type = "probs")


# Calculate precision, recall, and thresholds for each class

precision <- numeric()

recall <- numeric()

thresholds <- seq(0, 1, by = 0.01)

 

for (threshold in thresholds) {

  # Convert predicted probabilities to binary predictions based on threshold

  binary_predictions <- apply(probabilities, 2, function(x) ifelse(x >= threshold, 1, 0))

 

  # Calculate precision and recall for each class

  class_precision <- c()

  class_recall <- c()

  for (i in 1:ncol(binary_predictions)) {

    tp <- sum(binary_predictions[, i] == 1 & test_data$quality == levels(test_data$quality)[i])

    fp <- sum(binary_predictions[, i] == 1 & test_data$quality != levels(test_data$quality)[i])

    fn <- sum(binary_predictions[, i] == 0 & test_data$quality == levels(test_data$quality)[i])

    class_precision[i] <- tp / (tp + fp)

    class_recall[i] <- tp / (tp + fn)

  }

 

  # Take the mean precision and recall across all classes

  precision <- c(precision, mean(class_precision))

  recall <- c(recall, mean(class_recall))

}

 

# Plot Precision-Recall curve

library(ggplot2)

df <- data.frame(thresholds, precision, recall)

ggplot(df, aes(x = recall, y = precision)) +

  geom_line() +

  geom_point() +

  labs(x = "Recall", y = "Precision", title = "Precision-Recall Curve")
